country_event = {
	id = mr_tou_events.2 # The Marquisate of Provence
	title = mr_tou_events.2.t
	desc = mr_tou_events.2.d
	picture = DIPLOMACY_SIGNING_eventPicture
	
	is_triggered_only = yes
	
	option = {
		name = mr_tou_events.2.a # Restore the title across the Rhone.
		hidden_effect = {
			provence_area = {
				add_core = PRO
			}
		}
		if = { 
			limit = { NOT = { exists = PRO } }
			release = PRO
		}
		hidden_effect = {
			provence_area = {
				cede_province = PRO
			}
			every_province = {
				limit = {
					country_or_non_sovereign_subject_holds = ROOT
					area = savoy_dauphine_area
				}
				cede_province = PRO
				add_core = PRO
			}
		}
		custom_tooltip = tou_provence_recieves_gains
		create_march = PRO
		PRO = {
			add_government_reform = provencal_marquisate
		}
		custom_tooltip = " "
		west_burgundy_area = {
			add_permanent_claim = PRO
		}
		bourgogne_area = {
			add_permanent_claim = PRO
		}
	}

	option = {
		name = mr_tou_events.2.b # Provence must be integrated!
		if = {
			limit = {
				PRO = { is_subject_of = ROOT }
			}
			inherit = PRO
		}
		provence_area = {
			add_local_autonomy = -25
			add_nationalism = -5
		}
		custom_tooltip = " "
		west_burgundy_area = {
			add_permanent_claim = PRO
		}
		bourgogne_area = {
			add_permanent_claim = PRO
		}
	}
}
