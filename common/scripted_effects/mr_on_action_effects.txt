#for elite unit xp
elite_unit_xp_combat_effect = {
	set_variable = {
		which = base_xp
		value = 1
	}
	#find ratio or rajput/total FL
	export_to_variable = {
		variable_name = xp_num_of_rajput_var
		value = trigger_value:num_of_rajput
		who = ROOT
	}
	multiply_variable = {
		which = base_xp
		which = xp_num_of_rajput_var
	}
	export_to_variable = {
		variable_name = xp_army_size_var
		value = trigger_value:army_size
		who = ROOT
	}
	divide_variable = {
		which = base_xp
		which = xp_army_size_var
	}
	
	#########
	
	#so it doesnt divide by 0
	if = {
		limit = {
			NOT = {
				check_variable = {
					which = land_units_killed
					value = 0.01
				}
			}
		}
		set_variable = {
			which = land_units_killed_copy
			value = 0.01
		}
	}
	else = {
		set_variable = {
			which = land_units_killed_copy
			which = land_units_killed
		}
	}
	
	if = {
		limit = {
			NOT = {
				check_variable = {
					which = land_units_lost
					value = 0.01
				}
			}
		}
		set_variable = {
			which = land_units_lost_copy
			value = 0.01
		}
	}
	else = {
		set_variable = {
			which = land_units_lost_copy
			which = land_units_lost
		}
	}
	
	#ratio of battle
	divide_variable = {
		which = land_units_killed_copy
		which = land_units_lost_copy
	}
	
	#extra xp from casualties maxed out at 2x
	if = {
		limit = {
			check_variable = {
				which = land_units_killed_copy
				value = 2
			}
		}
		set_variable = {
			which = land_units_killed_copy
			value = 2
		}
	}
	
	multiply_variable = {
		which = base_xp
		which = land_units_killed_copy
	}
	
	#add to elite_unit_xp
	#elite_unit_xp initialized in generic_mission_events.50
	change_variable = {
		which = elite_unit_xp
		which = base_xp
	}
}
mr_maitreya_effect = {
	set_variable = {
		which = maitreyan_power_counter_cost
		value = 0
	}
	if = {
		limit = {
			has_ruler_flag = xiao_mingwang_flag
			religion = mr_maitreya
		}
		set_variable = {
			which = maitreyan_power_counter
			value = 100
		}
	}
	else_if = {
		limit = {
			religion = mr_maitreya
		}
		set_variable = {
			which = maitreyan_power_counter
			value = 0
		}
	}
	#Gain a base amount.
	if = {
		limit = {
			religion = mr_maitreya
		}
		export_to_variable = {
			which = ruler_adm_check
			value = ADM
		}
		export_to_variable = {
			which = ruler_dip_check
			value = DIP
		}
		export_to_variable = {
			which = ruler_mil_check
			value = MIL
		}
		set_variable = {
			which = ruler_check_gain
			value = 0
		}
		change_variable = {
			which = ruler_check_gain
			which = ruler_mil_check
		}
		change_variable = {
			which = ruler_check_gain
			which = ruler_dip_check
		}
		change_variable = {
			which = ruler_check_gain
			which = ruler_adm_check
		}
		while = {
			limit = {
				#While their ruler_score is more than 1, add 10 Mingwang Authority
				check_variable = {
					which = ruler_check_gain
					value = 1
				}
			}
			change_variable = {
				which = maitreyan_power_counter
				value = 5
			}
			subtract_variable = {
				which = ruler_check_gain
				value = 1
			}
		}
	}
}
