					culture = persian

					culture = turkish

					province_id = 412
					province_id = 2327

				region = horn_of_africa_region

			AND = {
				culture = al_misr_arabic
				NOT = { area = al_wahat_area }
				NOT = { area = cyrenaica_area }

				culture_group = hindusthani

				any_neighbor_province = {
					NOT = { owned_by = ROOT }

				}
			}
		}
		modifier = {
			factor = 0.15
			scale = yes
			force_potential = yes	# Hide the this modifier when potential is false
			potential = {
				owner = { has_estate = estate_nobles }

				tooltip = FEUDALISM_AND_NOBLES

				owner = {

					has_institution = feudalism
				}

firearms_institution = {

	bonus = {
		fire_damage = 0.05
	}

	trade_company_efficiency = 0.4
	
	effect = {
		if = {
			limit = {
				mil_tech = 1
				NOT = {  mil_tech = 2 }
			}
			add_mil_tech = 3
		}
		else_if = {
			limit = {
				mil_tech = 2
				NOT = {  mil_tech = 3 }
			}
			add_mil_tech = 2
		}
		else_if = {
			limit = {
				mil_tech = 3
				NOT = {  mil_tech = 4 }
			}
			add_mil_tech = 1
		}
		else_if = {
			limit = {
				mil_tech = 4
				NOT = {  mil_tech = 5 }
			}
			add_mil_tech = 1
		}
		else_if = {
			limit = {
				mil_tech = 5
				NOT = {  mil_tech = 6 }
			}
			add_mil_tech = 1
		}
	}
	
	#only used in bookmarks
	historical_start_date = 1455.1.1
	historical_start_province = 116 # Florence

	history = {
		is_year = 1350
		OR = {
			OR = {
				region = south_china_region
				region = xinan_region
				region = north_china_region
				region = korea_region

				region = malaya_region
				region = indo_china_region
				region = iberia_region
			}
			owner = {
				OR = {
					tag = POL
					tag = BLH
					tag = BOH
					tag = SOS
				}
			}
		}
	}

		always = yes

					firearms_institution = 100

			}
