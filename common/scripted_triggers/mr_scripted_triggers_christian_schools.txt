#Scholars from where?
#Country Scope
scholar_from_our_school_trigger_christian = {
	has_country_modifier = $school$_rite_modifier
	ROOT = {
		religious_school = {
			group = christian
			school = $school$_rite
		}
	}
}
our_scholar_matches_their_school_trigger_christian = {
	PREV = { has_country_modifier = $school$_rite_modifier }
	religious_school = {
		group = christian
		school = $school$_rite
	}
}
